\subsection*{Fiber Equations}

\textbf{Spatio-temporal membrane potential dynamics:}
\[
\frac{\partial^2 V_m}{\partial x^2}
- (r_i + r_e)c_m \frac{\partial V_m}{\partial t}
- \frac{r_i + r_e}{r_m} V_m
= r_e\, i_p
\]

\textbf{Equivalent form:}
\[
\frac{r_m}{r_i + r_e}\frac{\partial^2 V_m}{\partial x^2}
- r_m c_m \frac{\partial V_m}{\partial t}
- V_m
= r_e \frac{r_m}{r_i + r_e} i_p
\]

\subsubsection*{Space and Time Constants $\lambda, \tau$}

\textbf{Space constant}, dist. after which current changes with e:
\[
\lambda
= \sqrt{\frac{r_m}{r_i + r_e}}
\;\approx\;
\sqrt{\frac{r_m}{r_i}}
\;\overset{r_e \ll r_i}{=}\;
\sqrt{
\frac{R_m}{2\pi a}
\frac{\pi a^2}{R_i}
}
\]

\textbf{Time constant:}
\(
\tau = r_m c_m
\)\\
(Time for a current to decay with 1/e)

\subsubsection*{General Differential Equation}
\(\lambda^2 \frac{\partial^2 v_m}{\partial x^2} - \tau \frac{\partial v_m}{\partial t}- v_m = r_e \lambda^2 i_p\)

\subsubsection*{Steady-State Solution}
\(\lambda^2\frac{d^2v_m}{dx^2} -v_m = r_e\lambda^2i_p\) with \(i_p = I_0\delta (x)\) \\
The system is stimulated with a point source at x = 0. \\
\textbf{Homogeneous solution:} \(v_m = Ae^{-x/\lambda} + Be^{x/\lambda}\) \\
\textbf{Particular solution:} \(v_m = -\frac{r_e\lambda I_0}{2} e^{-|x|/\lambda}\) (t $\rightarrow \infty $)\\

\textbf{Observations:}
{\tiny
\begin{itemize}
    \item $|V_m|$ is maximal close to the stimulus and decays exponentially with distance
    \item $V_m \propto r_e$ (extracellular resistance)
    \item Positive current hyperpolarizes the membrane
    \item $\lambda$ determines the spatial extent of the stimulus along the fiber
\end{itemize}
}

\vfill
\subsubsection*{Time-Varying Response}

\textbf{Step stimulation:}
\[
i_p = I_0 \delta(x)\, u(t),
\qquad
\text{$u(t)$ = step function},
\;
(t = x = 0)
\]

{\tiny
\(
V_m(x,t)
= \frac{r_i \lambda I_0}{4}
\Bigg[
e^{-|x|/\lambda}
\Bigg(
1 - \mathrm{erf}
\!\left(
\frac{|x|}{2\lambda}\sqrt{\frac{\tau}{t}}
- \sqrt{\frac{t}{\tau}}
\right)
\Bigg)
-
e^{-|x|/\lambda}
\Bigg(
1 - \mathrm{erf}
\!\left(
\frac{|x|}{2\lambda}\sqrt{\frac{\tau}{t}}
+ \sqrt{\frac{t}{\tau}}
\right)
\Bigg)
\Bigg]
\)
}

\textbf{Error function:}\\
{\tiny
\(
\mathrm{erf}(x)
= \frac{2}{\sqrt{\pi}}
\int_{0}^{x} e^{-z^2}\, dz,
\qquad
\mathrm{erf}(0)=0,
\;
\mathrm{erf}(x \rightarrow \pm\infty)=\pm1
\)
}
