\subsection*{Diffusion}

\textbf{Concentration:} $c(x,t) \hfill [\text{M} = \frac{\text{mol}}{L} = \frac{1000\text{mol}}{m^3}]$\\
\textbf{Flux:} $\Phi(x,t) \hfill [\frac{\text{mol}}{cm^2 \cdot s} = 10^4 \cdot \frac{\text{mol}}{m^2 \cdot s}]$\\

% ----------------
\textbf{Fick's First Law (1D):}

\begin{minipage}{0.5\linewidth}
    \[
    \boxed{
    \phi(x,t)
    = -D \frac{\partial c(x,t)}{\partial x}
    }
    \]
\end{minipage}
\begin{minipage}{0.5\linewidth}
\vspace{0pt}
    \[
    \boxed{
    t_{1/2} \approx \frac{x_{1/2}^{\,2}}{D}
    }
    \]
\end{minipage}
\begin{minipage}[t]{1\linewidth}
\tiny
    $D$ : diffusion coefficient $[\frac{m^2}{s}]$ 
    is smaller for larger molecules and in viscous media \\
    $t_{1/2}$ : half-time (50\% conc.) $[s]$ \hfill $x_{1/2}$ : half-distance (50\% conc.) $[m]$ \\
\end{minipage}

\textbf{Flux through membrane (thickness $d$):}
\[
\boxed{
\Phi
= -D\,\frac{c_{out} - c_{in}}{d}
}
\]
\vfill \null \columnbreak
% ----------------
% ----------------
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Continuity equation:}
    \[
    \frac{\partial c(x,t)}{\partial t}
    = - \frac{\partial \phi(x,t)}{\partial x}
    \]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Diffusion equation (1D):}
    \[
    \frac{\partial c(x,t)}{\partial t}
    = D\,\frac{\partial^2 c(x,t)}{\partial x^2}
    \]
\end{minipage}
% ----------------


% ----------------
\textbf{Point source solution (Gaussian):}
\[
c(x,t)
= \frac{n_0}{\sqrt{4\pi Dt}}
\exp\!\left(-\frac{x^2}{4Dt}\right),
\qquad
\sigma = \sqrt{2Dt}
\]

% ----------------
\textbf{Steady state diffusion:}
\[
\frac{\partial c}{\partial t}=0
\quad \Rightarrow \quad
\frac{\partial^2 c}{\partial x^2}=0
\quad \Rightarrow \quad
c(x)=ax+b
\]
