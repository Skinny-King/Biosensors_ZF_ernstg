\subsection*{Ion Transport}
% ================================

\textbf{Nernst--Planck equation:}
\[
\boxed{
J_n
= -D_n\,\frac{dc_n}{dx}
- u_n z_n F c_n\,\frac{d\psi}{dx}
}
\]

% ----------------

\begin{minipage}[t]{0.51\linewidth}
\textbf{Nernst equilibrium potential:}
    \[
    \boxed{
    V_n
    = \frac{RT}{z_n F}
    \ln\!\left(\frac{c_{\text{out}}}{c_{\text{in}}}\right)
    }
    \]

    \[
    \boxed{
    V_n
    \approx \frac{60}{z_n}
    \log_{10}\!\left(\frac{c_{\text{out}}}{c_{\text{in}}}\right)
    }
    \]
\end{minipage}
\begin{minipage}[t]{0.51\linewidth}
\vspace{0pt}
\tiny
    $V_n$ : = $E_n$ Nernst potential $[V]$ \\
    $R$ : gas constant $[8.314\,\frac{J}{mol\cdot K}]$ \\
    $T$ : temperature $[K]$ \\
    $F$ : Faraday constant $[96485\,\frac{C}{mol}]$ \\
    $z_n$ : ion valence (e.g., $+1$ for Na$^+$) \\
    $c_{\text{in}}$, $c_{\text{out}}$ : \text{in}/outside ion concentration $[\text{M}]$ \\
    $G_n$ : membr. conduct. $[\frac{S}{m^2}=\frac{1}{\Omega \cdot m^2}]$\\
    $D_n$ : diffusion coeff. $[m^2/s]$ \\
\end{minipage}
\vspace{0.5em}


% ----------------
\textbf{Donnan equilibrium (partition rule):}
\[
\boxed{
\left(\frac{c_{\text{out}}}{c_{\text{in}}}\right)^{1/z}
= \text{const}
}
\]
\textbf{Electroneutrality:}
 \[
c_{K,\text{in}}(0) + c_{Na,\text{in}}(0)
=
c_{K,\text{in}}(t) + c_{Na,\text{in}}(t),
\qquad
{\tiny\text{in} = \text{intracell.}}
\]
