\subsection*{Multiple Permeant Ion Model}
\begin{minipage}[t]{0.51\linewidth}
    \[
    x = \frac{X}{A},
    \quad
    J_m = \frac{I_m}{A}
    \]
    \[
    \boxed{
    J_{n} = G_{n} (V_m - V_n)
    }
    \]

\end{minipage}
\begin{minipage}[t]{0.51\linewidth}
\vspace{0pt}
\tiny
    $x$ : intrinsic size $[.../m^2]$ \\
    $A$ : membrane surface area $[m^2]$ \\
    $n$ : ion species index \\
    $m$ : total membrane index \\
    $J_m$ : membrane current density $[A/m^2]$ \\
    $I_m$ : membrane current $[A]$ \\
    $G_n$ : ion conductance $[S/m^2]$ \\
    $V_m$ : membrane potential $[V]$ \\
    $V_n$ or $E_n$ : Nernst potential of ion $n$ $[V]$
\end{minipage}
\begin{minipage}[t]{0.35\linewidth}

for a steady state $V_m$ ($J_m = 0$):
    \[

Rest: $J_m = 0, \quad J_K, J_{Na}, J_o \neq  0\\
V_m = \text{const} \approx -65 [mV]$\\

Equilibrium: $J_K, J_{Na}, J_o = 0\\ V_m = V_K = V_{Na} = V_o$
