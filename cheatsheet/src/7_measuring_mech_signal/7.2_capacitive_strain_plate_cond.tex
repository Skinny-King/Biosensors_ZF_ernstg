\subsection*{Capacitive Strain - Parallel Plate Capacitor}
\subsubsection*{Lateral Strain}

\begin{minipage}[t]{0.5\linewidth}
    \textbf{Capacitance formula:}
    \[
    C
    =
    \frac{A\,\varepsilon}{d}
    =
    \frac{A\,\varepsilon_r\,\varepsilon_0}{d}
    \]
\end{minipage}
\begin{minipage}[t]{0.5\linewidth}
\tiny
$C$ : capacitance $\left[\mathrm{F}\right]$ \\
$A$ : electrode overlap area $\left[\mathrm{m^2}\right]$ \\
$d$ : plate separation $\left[\mathrm{m}\right]$ \\
$\varepsilon_r$ : relative permittivity $\left[-\right]$ \\
$\varepsilon_0$ : vacuum permittivity $\left[\mathrm{F/m}\right]$
\end{minipage}

\textbf{Gauge factor:}
\[
GF
=
\frac{\Delta C / C_0}{\Delta L / L_0}
=
\frac{\Delta C / C_0}{\varepsilon}
=
\frac{\varepsilon}{\varepsilon} \qquad
\boxed{
GF = 1
}
\]
% \vfill

\subsubsection*{Pressure Sensor}

\begin{minipage}[t]{0.48\linewidth}
\textbf{Applied pressure:}\\
\[
p
=
\frac{F}{A}
=
\sigma
=
E\,\varepsilon
\]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
\textbf{Capacitance under strain:}
\[
C(\varepsilon)
=
C_0\,
\frac{(1+\nu\,\varepsilon)^2}{1-\varepsilon}
\]
\end{minipage}
\textbf{Gauge factor (capacitive):}
\[
GF
=
\frac{\Delta C / C_0}{\varepsilon}
\;\approx\;
\frac{2\nu + 1}{1-\varepsilon}
\]
