\subsection*{Langmuir Model}

\textbf{Reversible binding reaction:}
Kinetically limited adsorption described by the law of mass action.

\textbf{Reaction kinetics:}
\[
\frac{d[AR]}{dt}
= k_{\mathrm{on}}[A][R]
- k_{\mathrm{off}}[AR]
\;\overset{\text{eq.}}{\Rightarrow}\;
k_{\mathrm{on}}[A][R]
= k_{\mathrm{off}}[AR]
\]

\textbf{Affinity constant:}
\[
K_A
= \frac{1}{K_D}
= \frac{k_{\mathrm{on}}}{k_{\mathrm{off}}}
= \frac{[AR]}{[A][R]}
\]

\textbf{Equilibrium surface coverage (Langmuir):}
\[
\Theta_A
= \frac{[AR]}{[R_{\mathrm{tot}}]}
= \frac{K_A[A]}{1 + K_A[A]}
\]

\textbf{Competitive binding:}
Analyte $A$ and background molecules $B$ compete for the same receptor sites.

\textbf{Receptor conservation:}
\[
[R_{\mathrm{tot}}]
= [R] + [AR] + [BR]
\]

\textbf{Fractional receptor coverage (competitive Langmuir):}
\[
\Theta_A
= \frac{K_A^{A}[A]}
{1 + K_A^{A}[A] + K_A^{B}[B]},
\;
\Theta_B
= \frac{K_A^{B}[B]}
{1 + K_A^{A}[A] + K_A^{B}[B]}
\]
