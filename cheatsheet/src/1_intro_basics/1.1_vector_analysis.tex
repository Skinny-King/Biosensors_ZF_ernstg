\subsection{Vector Analysis}

\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\textbf{Dot product:}
{\tiny
\[
    \vec{A} \cdot \vec{B} = AB \cos \theta
\]
}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}

\textbf{Cross product:}
\tiny
\[
\vec{a} \times \vec{b} = \vec{c}
\]
\[
\begin{pmatrix}
a_x \\
a_y \\
a_z
\end{pmatrix}
\times
\begin{pmatrix}
b_x \\
b_y \\
b_z
\end{pmatrix}
=
\begin{pmatrix}
a_y b_z - a_z b_y \\
a_z b_x - a_x b_z \\
a_x b_y - a_y b_x
\end{pmatrix}
\]
\end{minipage}

\begin{minipage}[t]{0.48\linewidth}
\vspace{2pt}

\textbf{Gradient operator:}
\tiny
\[
    \nabla
    \equiv
    \frac{\partial}{\partial x} \vec{e}_x
    + \frac{\partial}{\partial y} \vec{e}_y
    + \frac{\partial}{\partial z} \vec{e}_z
\]
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\linewidth}
\vspace{2pt}
\textbf{Divergence:}
\tiny
\[
    \nabla \cdot \vec{J}
    \equiv \mathrm{div}\,\vec{J}
    = \frac{\partial J_x}{\partial x}
    + \frac{\partial J_y}{\partial y}
    + \frac{\partial J_z}{\partial z}
\]
\end{minipage}
\textbf{Laplace equation:}
{\tiny
\[
    \nabla^2 \Psi
    = \frac{\partial^2 \Psi}{\partial x^2}
    + \frac{\partial^2 \Psi}{\partial y^2}
    + \frac{\partial^2 \Psi}{\partial z^2}
    = 0
    \qquad \text{if there are no sources}
\]
}
\textbf{The field of a current point source:}

\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\[
\fbox{$
    \Phi_e(r)
    = \frac{I_0}{4\pi\sigma}\,\frac{1}{r}
$}
\]
\end{minipage}\hfill
\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\tiny
\raggedright
$\Phi_e$ : electric potential \\
$I_0$    : source current \\
$\sigma$ : conductivity \\
$r$      : distance from the source
\end{minipage}

\textbf{Dirac delta function:}

{\tiny
\[
\begin{aligned}
\nabla^2 \left( \frac{1}{r} \right)
&= -4\pi \delta(r)
\qquad
\int \delta(r)\, dv = 1
\end{aligned}
\]
}
