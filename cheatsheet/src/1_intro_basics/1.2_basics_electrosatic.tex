\subsection*{Electrostatic}

\begin{minipage}[t]{0.58\linewidth}
\textbf{The field of a current point source:}
\vspace{-3pt}
\[
\fbox{$
    \Phi_e(r)
    = \frac{I_0}{4\pi\sigma}\,\frac{1}{r}
$}
\]

\end{minipage}\hfill
\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\tiny
\raggedright
$\Phi_e$ : electric potential $[V]$\\
$I_0$    : point source current $[A]$ \\
$\sigma$ : conductivity $[\frac{1}{\Omega m}]$\\
$r$      : distance from the source $[m]$\\
$\vec{E}$: electric field $[\frac{V}{m}]$ \\
$I_v$    : point sourcecurrent density $[\frac{A}{m^3}]$ \\
\end{minipage}

\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}

\textbf{Electrostatic field:}
\[
    \vec{E} 
    = 
    -\nabla \Phi
    =
    \frac{I_0}{4\pi r^2\sigma}\mathbf{e_r}\ 
\]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
\vspace{-1pt}
\textbf{Current density:} 
\vspace{-1pt}
\[
    \vec{J} 
    =
    \sigma \vec{E}
    =
    -\sigma \nabla \Phi
    =
    \frac{I_0}{4\pi r^2}\mathbf{e_r}\
\]
\end{minipage}
\begin{minipage}[t]{\linewidth}
\vspace{0pt}
\textbf{Poisson equation and outflow of current from} $I_v$ \textbf{ point source:}
\[
\nabla^2 \Phi 
= -\frac{I_v}{\sigma} \qquad
\Longleftrightarrow \qquad
\boxed{
I_v 
= \nabla \cdot \vec{J}
= -\sigma \nabla^2 \Phi
}
\]
\end{minipage}
