\subsection*{Capacitance and Resistance}

\begin{minipage}[t]{0.6\linewidth}
\vspace{0pt}
\textbf{Parallel plate capacitor:}
    \[
    C \equiv \frac{Q}{V} = \frac{\varepsilon A}{d}
    \]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\tiny
    $C$ : capacitance $[F]$ \\
    $Q$ : charge $[C]$ \\
    $V$ : voltage $[V]$ \\
    $\varepsilon$ : dielectric permittivity $[\frac{F}{m}]$ \\
    $A$ : area $[m^2]$ \\
    $d$ : distance $[m]$ \\
\end{minipage}
\normalsize
\begin{minipage}[t]{0.6\linewidth}
\vspace{-8pt}
\textbf{Resistance and resistivity:}
    \[
    R = \rho \frac{L}{A_x} = \rho \frac{L}{a^2}, \qquad \rho = \frac{1}{\sigma}
    \]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
\vspace{0pt}
\tiny
    $R$ : resistance $[\Omega]$ \\
    $\rho$ or $R_i$ : resistivity $[\Omega\cdot m]$ \\
    $L$ : length $[m]$ \\
    $A_x$ : cross-section area $[m^2]$ \\
    $\sigma$ : conductivity $[\mathrm{S/m}]$ \\
\end{minipage}
