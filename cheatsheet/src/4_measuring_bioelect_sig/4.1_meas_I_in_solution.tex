\begin{minipage}[t]{0.48\linewidth}
    \tiny
    $u_{ion}$ : mobility of ions $\left[\mathrm{m^2/(V\,s)}\right]$ \\
    $u_{\mathrm{Cl}}$ : mobility of Cl$^-$ ions $\left[\mathrm{m^2/(V\,s)}\right]$ \\
    $D_{ion}$ : diffusion coeff. of ion$\left[\mathrm{m^2/s}\right]$ \\
    $Z_{ion}$ : valence of ion $[-]$ \\
    $\sigma$ : electrical conductivity $\left[\mathrm{S/m}\right]$ \\
    $C_{\mathrm{KCl}}$ : concentration of KCl $\left[\mathrm{mol/l}\right]$
\end{minipage}
