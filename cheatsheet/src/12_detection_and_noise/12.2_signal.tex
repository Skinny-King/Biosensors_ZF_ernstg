\subsection*{Signal}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Signal current:}
    \[
    \langle I_{\text{sig}} \rangle
    = \frac{q\,\eta\,\bar{n}}{\Delta t}
    = q\,\eta\, n
    \]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Optical power:}
    \[
    P
    = \bar{n}\,\frac{hc}{\lambda\,\Delta t}
    = n\,\frac{hc}{\lambda}
    \]
\end{minipage}

\begin{minipage}[t]{0.4\linewidth}
    \textbf{Responsivity:}
    \[
    \mathcal{R}
    = \frac{\langle I \rangle}{P}
    = \frac{\eta\, q\, \lambda}{hc}
    \]
\end{minipage}
\begin{minipage}[t]{0.6\linewidth}
{\tiny
$q$ : elementary charge $\left(1.6\times10^{-19}\,\mathrm{C}\right)$ \\
$\eta$ : quantum efficiency $[-]$ \\
$\bar{n}$ : number of incident photons in $\Delta t$ \\
$n$ : photon rate $\left[\mathrm{s^{-1}}\right]$ \\
$\lambda$ : wavelength $\left[\mathrm{m}\right]$
}
\end{minipage}
