\subsection*{Lenses}
\textbf{Positive lens:}
At least one convex surface; converges rays; $f>0$.

\textbf{Negative lens:}
At least one concave surface; diverges rays; $f<0$.

\textbf{Real image:}
Formed by actual ray intersection; can be projected onto a screen or sensor.

\textbf{Virtual image:}
Formed by apparent ray intersection; cannot be projected onto a screen.
\textbf{Thin lens equation:}
\[
\frac{1}{f}
= \frac{1}{u} + \frac{1}{v}
= (n-1)\!\left(\frac{1}{R_1}-\frac{1}{R_2}\right)
\]

\begin{minipage}[t]{0.6\linewidth}

\textbf{Magnification:}
\[
M = \frac{h_i}{h_o}
= -\frac{v}{|u|}
= \left|\frac{f_2}{f_1}\right|
\]

\[
(M<0:\ \text{real, inverted image})
\]

\textbf{Dioptre:}
\[
D = \frac{1}{f}
\]
\end{minipage}
\begin{minipage}[t]{0.3\linewidth}
\vspace{0pt}
\tiny
$f$ : focal length $\left[\mathrm{m}\right]$ \\
$u$ : object distance $\left[\mathrm{m}\right]$ \\
$v$ : image distance $\left[\mathrm{m}\right]$ \\
$R_1, R_2$ : radii of curvature of lens surfaces $\left[\mathrm{m}\right]$ \\
$n$ : refractive index $[-]$ \\
$h_o$ : object height $\left[\mathrm{m}\right]$ \\
$h_i$ : image height $\left[\mathrm{m}\right]$ \\
$M$ : magnification $[-]$ \\
$D$ : dioptre $\left[\mathrm{m^{-1}}\right]$
\end{minipage}

\subsubsection*{Ray-Tracing}
\begin{itemize}
    \item Parallel rays are focused at the focal point.
    \item Rays through the focal point exit parallel to the optical axis.
    \item Rays through the lens center are undeviated.
\end{itemize}
