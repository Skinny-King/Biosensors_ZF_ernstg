\subsection*{Diffusion}

\textbf{Concentration:} $c(x,t) \hfill [\text{M} = \frac{\text{mol}}{L} = \frac{1000\text{mol}}{m^3}]$\\
\textbf{Flux:} $\Phi(x,t) \hfill [\frac{\text{mol}}{cm^2 \cdot s} = 10^4 \cdot \frac{\text{mol}}{m^2 \cdot s}]$\\

% ----------------
\textbf{Fick's First Law (1D):}

\begin{minipage}{0.5\linewidth}
    \[
    \boxed{
    \phi(x,t)
    = -D \frac{\partial c(x,t)}{\partial x}
    }
    \]
\end{minipage}
\begin{minipage}{0.5\linewidth}
\vspace{0pt}
    \[
    \boxed{
    t_{1/2} \approx \frac{x_{1/2}^{\,2}}{D}
    }
    \]
\end{minipage}
\begin{minipage}[t]{1\linewidth}
\tiny
    $D$ : diffusion coefficient $[\frac{m^2}{s}]$ 
    is smaller for larger molecules and in viscous media \\
    $t_{1/2}$ : half-time (50\% conc.) $[s]$ \hfill $x_{1/2}$ : half-distance (50\% conc.) $[m]$ \\
\end{minipage}

\textbf{Flux through membrane (thickness $d$):}
\[
\boxed{
\Phi
= -D\,\frac{c_o - c_i}{d}
}
\]
\vfill \null \columnbreak
% ----------------
% ----------------
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Continuity equation:}
    \[
    \frac{\partial c(x,t)}{\partial t}
    = - \frac{\partial \phi(x,t)}{\partial x}
    \]
\end{minipage}
\begin{minipage}[t]{0.48\linewidth}
    \textbf{Diffusion equation (1D):}
    \[
    \frac{\partial c(x,t)}{\partial t}
    = D\,\frac{\partial^2 c(x,t)}{\partial x^2}
    \]
\end{minipage}
% ----------------


% ----------------
\textbf{Point source solution (Gaussian):}
\[
c(x,t)
= \frac{n_0}{\sqrt{4\pi Dt}}
\exp\!\left(-\frac{x^2}{4Dt}\right),
\qquad
\sigma = \sqrt{2Dt}
\]

% ----------------
\textbf{Steady state diffusion:}
\[
\frac{\partial c}{\partial t}=0
\quad \Rightarrow \quad
\frac{\partial^2 c}{\partial x^2}=0
\quad \Rightarrow \quad
c(x)=ax+b
\]

\subsection*{Ion Transport}
% ================================

\textbf{Nernst--Planck equation:}
\[
\boxed{
J_n
= -D_n\,\frac{dc_n}{dx}
- u_n z_n F c_n\,\frac{d\psi}{dx}
}
\]

% ----------------

\begin{minipage}[t]{0.51\linewidth}
\textbf{Nernst equilibrium potential:}
    \[
    \boxed{
    V_n
    = \frac{RT}{z_n F}
    \ln\!\left(\frac{c_o}{c_i}\right)
    }
    \]

    \[
    \boxed{
    V_n
    \approx \frac{60}{z_n}
    \log_{10}\!\left(\frac{c_o}{c_i}\right)
    \ [mV]
    }
    \]
\end{minipage}
\begin{minipage}[t]{0.51\linewidth}
\vspace{0pt}
\tiny
    $V_n$ : Nernst potential $[mV]$ \\
    $R$ : gas constant $[8.314\,\frac{J}{mol\cdot K}]$ \\
    $T$ : temperature $[K]$ \\
    $F$ : Faraday constant $[96485\,\frac{C}{mol}]$ \\
    $z_n$ : ion valence (e.g., $+1$ for Na$^+$) \\
    $c_o$, $c_i$ : outside/inside ion concentration $[\text{M}]$ \\
    $G_n$ : membrane conductance $[\frac{S}{m^2}=\frac{1}{\Omega \cdot m^2}]$
\end{minipage}
\vspace{0.5em}

% ----------------
\textbf{Membrane potential (multiple ions, steady state):}
    \[
    \boxed{
    V_m
    = \frac{G_{Na}V_{Na} + G_K V_K + G_o V_o}
    {G_{Na}+G_K+G_o}
    }
    \]
% ----------------
\textbf{Donnan equilibrium (partition rule):}
\[
\boxed{
\left(\frac{c_o}{c_i}\right)^{1/z}
= \text{const}
}
\]
% ----------------
\subsection*{Resting Potential and Ion Concentrations}

\begin{itemize}
    \item $[K^+]_i > [K^+]_o$
    \item $[Na^+]_i < [Na^+]_o$
    \item $[K^+]_i \uparrow \;\Rightarrow\; V_m \downarrow$
    \item $[K^+]_o \uparrow \;\Rightarrow\; V_m \uparrow$
    \item $[K^+]_i \approx [K^+]_o \;\Rightarrow\; V_m \approx 0$
    \item Effect of changes in $[Na^+]$ $<$ effect of changes in $[K^+]$
\end{itemize}

\textbf{Multiple permeant ion model:}
\[
J_n = G_n (V_m - V_n),
\qquad
J_m = \sum_n J_n
\]

\[
\boxed{
J_m = 0
\;\Rightarrow\;
V_m^0
= \sum_n \frac{G_n}{\sum_k G_k}\,V_n
}
\]
